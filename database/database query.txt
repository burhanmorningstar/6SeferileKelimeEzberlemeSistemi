CREATE DATABASE 6kelimetekrar;

CREATE TABLE users (
    user_id INT AUTO_INCREMENT PRIMARY KEY,
    user_password VARCHAR(255) NOT NULL,
    user_email_address VARCHAR(45) NOT NULL UNIQUE,
    user_fullname VARCHAR(255) NOT NULL,
    verification_code VARCHAR(255)
);

CREATE TABLE words (
    word_id INT AUTO_INCREMENT PRIMARY KEY,
    word VARCHAR(45) NOT NULL,
    word_meaning VARCHAR(45) NOT NULL,
    word_in_sentence VARCHAR(255),
    word_image VARCHAR(45),
    word_voice VARCHAR(255)
);

CREATE TABLE settings (
    setting_id INT AUTO_INCREMENT PRIMARY KEY,
    user_id INT,
    word_limit INT,
    FOREIGN KEY (user_id) REFERENCES users(user_id)
);

CREATE TABLE knownWords (
    known_word_id INT AUTO_INCREMENT PRIMARY KEY,
    user_id INT,
    word_id INT,
    last_asked_date DATE,
    FOREIGN KEY (user_id) REFERENCES users(user_id),
    FOREIGN KEY (word_id) REFERENCES words(word_id)
);

CREATE TABLE wordDetails (
    id INT AUTO_INCREMENT PRIMARY KEY,
    word_counter INT,
    word_id INT,
    user_id INT,
    last_asked_date DATE,
    next_quiz_date DATE,
    how_many_times_asked INT,
    how_many_wrong_answers INT,
    how_many_correct_answers INT,
    FOREIGN KEY (user_id) REFERENCES users(user_id),
    FOREIGN KEY (word_id) REFERENCES words(word_id)
);
