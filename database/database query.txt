CREATE TABLE users (
  user_id SERIAL PRIMARY KEY,
  user_password VARCHAR(255),
  user_email_address VARCHAR(45),
  user_fullname VARCHAR(255)
);

CREATE TABLE words (
  word_id SERIAL PRIMARY KEY,
  word_meaning VARCHAR(45),
  word VARCHAR(45),
  word_image VARCHAR(45),
  word_in_sentence VARCHAR(255),
  word_voice VARCHAR(255)
);

CREATE TABLE wordDetails (
  id SERIAL PRIMARY KEY,
  word_counter INT,
  word_id INT REFERENCES words(word_id),
  user_id INT REFERENCES users(user_id)
);
